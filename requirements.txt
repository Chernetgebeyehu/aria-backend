# ARIA Backend — Python Dependencies
# =====================================
# Install all of these with: pip install -r requirements.txt
#
# Why pinned versions?
# Without pins, pip might install newer versions that have breaking changes.
# Pins = "use exactly this version" = your server works the same everywhere.

# ─── CORE SERVER ──────────────────────────────────────────────────────────────
fastapi==0.111.0          # The web framework
uvicorn[standard]==0.29.0 # The server engine that runs FastAPI
httpx==0.27.0             # Makes HTTP requests to Gemini/OpenRouter
python-dotenv==1.0.1      # Reads .env file for secrets
pydantic==2.7.0           # Validates request/response data shapes

# ─── MEMORY SYSTEM (Phase 3) ──────────────────────────────────────────────────
# These are needed for long-term memory.
# If Railway build fails due to these, remove them temporarily
# and the server will run without memory (main.py handles this gracefully).
chromadb==0.5.0                  # Vector database — stores and searches embeddings
sentence-transformers==3.0.0     # Converts text to embedding vectors
torch==2.3.0                     # PyTorch — required by sentence-transformers
